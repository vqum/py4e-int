class Book:
    """ Represents a book object """

    def __init__(self, title, author):
        self.title = title
        self.author = author

    def __str__(self):
        return "title: " + self.title + " author: " + self.author

    def get_author(self):
        return self.author

def main():
     b2 = Book("A Wrinkle in Time", "M. L'Engle")
     print(b2)
     b1 = Book("Goodnight Moon", "Margaret Wise Brown")
     print(b1)
